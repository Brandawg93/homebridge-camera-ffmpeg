{
    "platform":"Camera-ffmpeg",
    "cameras":[
      {
          "name":"Maygion",
          "videoConfig":{
            "source":"-re -i rtsp://[USER]:[PASSWORD]@ip-address:81/videostream.cgi",

            "maxStreams":2,
            "maxWidth":1280,
            "maxHeight":720,
            "maxFPS":16,
            "vcodec":"h264_omx"
          }
      }
    ]
}
